S -> INSTRUCCION

INSTRUCCION -> COMENTARIO
			 | EXEC PATH
			 | PAUSE
			 | MKDISK MKDISKPARAMS
			 | RMDISK PATH
			 | FDISK FDISKPARAM
			 | MOUNT MOUNTPARAMS		 
			 | UNMOUNT UNMOUNTPARAMS
			 | MKFS MKFSPARAMS	 
			 | LOGIN LOGINPARAMS	 
			 | LOGOUT
			 | MKGRP MKGRPPARAMS
			 | RMGRP RMGRPPARAMS	 
			 | MKUSR MKUSRPARAMS
			 | RMUSR RMUSRPARAMS
			 | CHMOD CHMODPARAMS

			 | MKFILE MKFILEPARAMS
			 | CAT CATPARAMS
			 | RM RMPARAMS
			 | EDIT EDITPARAMS
			 | REN RENPARAMS
			 | MKDIR MKDIRPARAMS
			 | CP CPPARAMS
			 | MV MVPARAMS
			 | FIND FINDPARAMS
			 | CHOWN CHOWNPARAMS
			 | CHGRP CHGRPPARAMS
			 | LOSS
			 | RECOVERY
			 | REP REPPARAMS

PATH -> - path - > (ruta | cadena)
			 
MKDISKPARAM -> - size - > entero
			 | PATH
			 | - name - > id . dsk
			 | - unit - > id
			
FDISKPARAM -> - size - > entero
			| - unit - > id
			| - tipo - > id
			| - fit - > id
			| - delete - > id
			| - name - > id 
			| - add - > entero
			| PATH
			
MOUNTPARAMS -> - name - > id 
			 | PATH
	
UNMOUNTPARAMS -> - idT num - > id 

MKFSPARAMS -> - idT - > id
			| - type - > id
			| - add - > entero
			| - unit - > id
			
LOGINPARAMS -> - usr - > (id | cadena)
			 | - pwd - > (id | cadena)
			 | - idT - > id

MKGRPPARAMS -> - idT - > id
			 | - name - > (id | cadena) 

RMGRPPARAMS -> - idT - > id
			 | - name - > (id | cadena) 

MKUSRPARAMS -> - usr - > (id | cadena)
			 | - pwd - > (id | cadena)
			 | - idT - > id
			 | - grp - > (id | cadena)

RMUSRPARAMS -> - usr - > (id | cadena)
			 | - idT - > id 

CHMODPARAMS -> - idT - > id
			 | PATH
			 | - ugo - > entero
			 | - R
